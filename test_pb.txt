package main

import (
	"fmt"
	"net"
	"net/http"
	"os"
	"time"
)

func main() {
	client := http.Client{
		Timeout: time.Millisecond * 2,
	}
	res, err := client.Head("http://example.com")
	if e, ok := err.(net.Error); ok && e.Timeout() {
		fmt.Println("timeout bitch!!")
		return
	} else if err != nil {
		fmt.Println(err)
		os.Exit(1)
	}
	defer res.Body.Close()
	fmt.Println("OK!")
}
